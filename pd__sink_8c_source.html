45static void send_request(pd_sink_t* sink, int object_position, uint32_t operating_current_ma) {
46 pd_packet_t packet;
47 packet.header = pd_header_build(1, 0x2, false, false, 2, sink->message_id_counter);
48 packet.num_data_objects = 1;
49
50 packet.data[0] = (object_position << 28) | (1 << 27) | (0 << 25) | (0 << 24) |

74void sink_handle_packet(pd_sink_t* sink, const pd_packet_t* packet) {
75 uint16_t message_type = packet->header & 0x1F;
76 uint8_t message_id = (packet->header >> 9) & 0x7;
77
78 if (sink->state == SINK_STATE_READY) {
79 if (message_type == 0x1) { // Source_Capabilities
80 send_good_crc(sink, message_id);
81 sink->source_capabilities = *packet;
82
83 if (sink->desired_voltage_mv > 0) {
84 int best_object_position = -1;
85 uint32_t best_voltage_mv = 0xffffffff;
86 uint32_t best_current_ma = 0xffffffff;
87
88 for (int i = 0; i < packet->num_data_objects; ++i) {
89 uint32_t pdo = packet->data[i];
90 if ((pdo >> 30) == 0) { // Fixed supply
91 uint32_t voltage_mv = ((pdo >> 10) & 0x3FF) * 50;
92 uint32_t current_ma = (pdo & 0x3FF) * 10;
93
94 if (voltage_mv >= sink->desired_voltage_mv && current_ma >= sink->desired_current_ma) {
95 if (best_object_position == -1 ||
96 (voltage_mv < best_voltage_mv) ||
97 (voltage_mv == best_voltage_mv && current_ma < best_current_ma)) {
98 best_object_position = i + 1;
99 best_voltage_mv = voltage_mv;
100 best_current_ma = current_ma;
101 }
102 }
103 }
104 }
105
106 if (best_object_position != -1) {
107 send_request(sink, best_object_position, sink->desired_current_ma);
108 }
109 }
110 }
111 } else if (sink->state == SINK_STATE_REQUESTING) {
112 if (message_type == 0x2) { // Accept
113 send_good_crc(sink, message_id);
114 if (sink->requested_object_position != -1) {
115 uint32_t pdo = sink->source_capabilities.data[sink->requested_object_position - 1];
116 sink->active_voltage_mv = ((pdo >> 10) & 0x3FF) * 50;
117 sink->active_current_ma = (pdo & 0x3FF) * 10;
118 }
119 sink->state = SINK_STATE_READY;
120 } else if (message_type == 0x3) { // Reject
121 send_good_crc(sink, message_id);
122 sink->state = SINK_STATE_READY;
123 }
