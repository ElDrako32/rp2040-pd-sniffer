105uint32_t hal_dma_read(uint32_t* buffer, uint32_t len) {
106 uint32_t count = 0;
107 while (read_pos != write_pos && count < len) {

175unsigned int hal_init(unsigned int port) {

239bool hal_get_packet(unsigned int port, struct pd_packet_s* packet) {
240 if (ports[port].read_pos != ports[port].write_pos) {
241 uint32_t data[1];
242 data[0] = ports[port].dma_buffer[ports[port].read_pos];
243 ports[port].read_pos = (ports[port].read_pos + 1) % DMA_BUFFER_SIZE_PER_PORT;
244 pd_phy_decode_stream(&ports[port].decoder, data, 1, packet_callback, (uintptr_t)port);
245 }
246 if (packet_received[port]) {
247 memcpy(packet, &packet_buffer[port], sizeof(pd_packet_t));
248 packet_received[port] = false;
249 return true;
250 }
251 return false;
252}
