55static void generate_crc_table(void) {
56 for (uint32_t i = 0; i < 256; i++) {
57 uint32_t c = i;
58 for (int j = 0; j < 8; j++) {
59 if (c & 1) {
60 c = 0xEDB88320L ^ (c >> 1);
61 } else {
62 c = c >> 1;
63 }
64 }
65 crc_table[i] = c;
66 }
67}

69uint32_t pd_crc32(const uint8_t *data, size_t len) {
70 if (!crc_table_generated) {
71 generate_crc_table();
72 crc_table_generated = true;
73 }
74
75 uint32_t crc = 0xFFFFFFFF;
76 for (size_t i = 0; i < len; i++) {
77 crc = (crc >> 8) ^ crc_table[(crc ^ data[i]) & 0xFF];
78 }
79 return ~crc;
80}

111void pd_encode_packet(const pd_packet_t* packet, uint32_t* encoded_data, size_t* encoded_len) {
112 uint32_t stream[64];
113 int stream_len = 0;
114
115 // SOP
116 stream[stream_len++] = K_CODE_SYNC1;
117 stream[stream_len++] = K_CODE_SYNC1;
118 stream[stream_len++] = K_CODE_SYNC2;
119 stream[stream_len++] = K_CODE_SYNC2;
120
121 append_4b5b(stream, &stream_len, packet->header);
122 for (int i = 0; i < packet->num_data_objects; ++i) {
123 append_4b5b_u32(stream, &stream_len, packet->data[i]);
124 }
125
126 uint8_t crc_data[2 + packet->num_data_objects * 4];
127 memcpy(crc_data, &packet->header, 2);
128 if (packet->num_data_objects > 0) {
129 memcpy(crc_data + 2, packet->data, packet->num_data_objects * 4);
130 }
131 uint32_t crc = pd_crc32(crc_data, 2 + packet->num_data_objects * 4);
132 append_4b5b_u32(stream, &stream_len, crc);
133
134 stream[stream_len++] = 0b01110; // EOP
135
136 // BMC encode
137 *encoded_len = 0;
138 int bit_count = 0;
139 uint64_t bmc_stream = 0;
140 int last_bit = 1;
141 uint64_t preamble = 0xAAAAAAAAAAAAAAAB;
142
143 for (int i = 0; i < 64; ++i) {
144 uint32_t bit = (preamble >> i) & 1;
145 if (bit == last_bit) {
146 bmc_stream |= (1ULL << bit_count);
147 bit_count++;
148 }
149 bmc_stream |= ((uint64_t)bit << bit_count);
150 bit_count++;
151 last_bit = bit;
152 if (bit_count >= 32) {
153 encoded_data[*encoded_len] = bmc_stream;
154 (*encoded_len)++;
155 bmc_stream >>= 32;
156 bit_count -= 32;
157 }
158 }
159
160 for (int i = 0; i < stream_len; ++i) {
161 for (int j = 4; j >= 0; --j) {
162 uint32_t bit = (stream[i] >> j) & 1;
163 if (bit == last_bit) {
164 bmc_stream |= (1ULL << bit_count);
165 bit_count++;
166 }
167 bmc_stream |= ((uint64_t)bit << bit_count);
168 bit_count++;
169 last_bit = bit;
170
171 if (bit_count >= 32) {
172 encoded_data[*encoded_len] = bmc_stream;
173 (*encoded_len)++;
174 bmc_stream >>= 32;
175 bit_count -= 32;
176 }
177 }
178 }
179 if (bit_count > 0) {
180 encoded_data[*encoded_len] = bmc_stream;

182 }
